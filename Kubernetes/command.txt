minikube start/delete  // start or delete minikube cluster
minikube status        // check status of minikube cluster
minikube stop         // stop minikube cluster

minikube dashboard    // open minikube dashboard in browser
minikube addons list  // list all available addons
minikube ip            // get minikube cluster ip address

kubectl create deployment --image=<image>    // create a deployment
kubectl get deployment       // get all deployments
kubectl get pots             // get all pods
kubectl delete deployment      // delete a deployment


kubectl expose deployment my-app --type=LoadBalancer --port=8080 // expose deployment as a service
minikube service my-app   // access the service in browser 
kubectl ger services     // get all services
kubectl delete service my-app  // delete a service
kubectl set image deployment my-react3 personal=pratik32/personal:04 // change image of running pod

kubectl rollout status deployment my-react3 // check status of rollout
kubectl rollout history deployment my-react3 // check history of rollout
kubectl rollout undo deployment my-react3 // undo to previous version of deployment
kubectl get all // get all resources

kubectl scale deployment my-express --replicas=4 // scale deployment to 4 replicas

kubectl apply -f k8s-deployment.yml // create deployment from yaml file
kubectl delete -f k8s-deployment.yml // delete deployment from yaml file
